{{/* Renders a element with WebP/JPG variants auto-generated by Hugo from a lossless source image. Browsers pick the best format they support. Parameters: .path — path to the source image, relative to /assets/ (e.g. "img/foo.png") .alt — alt text for the element */}} {{- $img := resources.Get .path -}} {{- if not $img -}} {{- errorf "picture partial: resource not found: %q" .path -}} {{- end -}} {{- $webp := $img.Process "webp" -}} {{- $jpg := $img.Process "jpg" -}}
